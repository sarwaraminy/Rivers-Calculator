SELECT 
    (Select Count(*) From RIVERS As B Where B.M3_PER_SECOND >= BRS.M3_PER_SECOND) As Row,
    BRS.M3_PER_SECOND
FROM 
    RIVERS AS BRS
ORDER BY 
    BRS.M3_PER_SECOND DESC;

SELECT BRS.M3_PER_SECOND AS 'Decreasion', 
    (Select Count(*) From RIVERS As B Where B.M3_PER_SECOND >= BRS.M3_PER_SECOND) As Row
    
FROM 
    RIVERS AS BRS
ORDER BY
